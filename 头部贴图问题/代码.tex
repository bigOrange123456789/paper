		\begin{algorithmic}[1]  
			
			\For{$i=0$; $i<height(A)$; $i++$ }  
			
			\For{$j=0$; $j<width(A)$; $j++$ }  
			
			\State $max=-Infinity;$
			
			\ForAll {$k$}  
			
			
			
			\label{code:TrainBase:getc}  
			\State $A[i][j]=k;$
			\State $f=f(A);$  
			
			\If{$f>max$ }  
			\State $max=f;$ 
			\State $maxK=k;$ 
			\EndIf
			
			\label{code:TrainBase:pos}  
			\EndFor; %\ForAll {$k$} 
			
			\State $A[i][j]=maxK$;
			
			\EndFor
			
			\EndFor  
			
			
			
			\label{code:recentEnd}  
		\end{algorithmic}  
	\end{algorithm}
